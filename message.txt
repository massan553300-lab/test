function experiment_plactice_first(p) {
  // おまじない
  let trial = jsPsych.getCurrentTrial();
  // これはタイムラインのデータを拾う（使い方はjsPsychのサイトを参照）。
  let angle = jsPsych.timelineVariable("angle", true);
  let colour = jsPsych.timelineVariable("colour", true);


  p.setup = function () {
    //刺激のデータ
    trial.data.angle = angle
    trial.data.colour = colour
    
    p.createCanvas(Canvaswidth,Canvasheight);
    start_length = p.random(-20,20)
    effect_size = start_length
    line_length = 150*1.5
    line_depart = 100*1.5
    p.angle = angle/90
    nw0 = p.millis();
    p.noCursor()
  }

  p.draw = function() {
    p.background(220);
    line_length_set = [line_length+effect_size,
      line_length,
      line_length]
    p.strokeWeight(4)
    p.translate(p.width/2, p.height/2)
    p.push()
    p.rotate(p.PI*2/4*p.angle)
    for (i = 0; i < 3; i++) {
      p.push()
      p.rotate(p.PI*2/3*i)
      p.stroke(colour[i])
      p.line(-line_length_set[i]/2, line_depart, line_length_set[i]/2, line_depart)
      p.pop()
    }
    p.pop()

    if (p.keyIsPressed) {
      if (p.key == 'j') {
        effect_size++
      }
      if (p.key == 'k') {
        effect_size--
      }
    }
  }
  p.keyTyped = function() {
    if (p.key === ' ') {
      trial.data.start_length = start_length
      trial.data.effect_size = effect_size
      trial.data.rt = p.millis() - nw0;
      trial.end_trial();
    };
  };
};
